     * @param markdown

     */
public static String markdownToHtml(String markdown) {
    Parser parser = Parser.builder().build();
    Node document = parser.parse(markdown);
    HtmlRenderer renderer = HtmlRenderer.builder().build();
    return renderer.render(document);
}

/**
     * 增加扩展[标题锚点，表格生成]
     * Markdown转换成HTML
     * @param markdown
     * @return
     */
public static String markdownToHtmlExtensions(String markdown) {
    //h标题生成id
    Set<Extension> headingAnchorExtensions = Collections.singleton(HeadingAnchorExtension.create());
    //转换table的HTML
    List<Extension> tableExtension = Arrays.asList(TablesExtension.create());
    Parser parser = Parser.builder()
        .extensions(tableExtension)
        .build();
    Node document = parser.parse(markdown);
    HtmlRenderer renderer = HtmlRenderer.builder()
        .extensions(headingAnchorExtensions)
        .extensions(tableExtension)
        .attributeProviderFactory(new AttributeProviderFactory() {
            public AttributeProvider create(AttributeProviderContext context) {
                return new CustomAttributeProvider();
            }
        })
        .build();
    return renderer.render(document);
}

/**
     * 处理标签的属性
     */
static class CustomAttributeProvider implements AttributeProvider {
    @Override
    public void setAttributes(Node node, String tagName, Map<String, String> attributes) {
        //改变a标签的target属性为_blank
        if (node instanceof Link) {
            attributes.put("target", "_blank");
        }
        if (node instanceof TableBlock) {
            attributes.put("class", "ui celled table");
        }
    }
}

      public void saveComment(Comment comment) {
    Long parentCommentId = comment.getParentComment().getId();
    if(parentCommentId != -1){
        //有父级，是回复的操作
        comment.setParentComment(commentRepository.findOne(parentCommentId));//设置父级关系
    }else{
        //没有父级，需要赋值一个空对象，否则-1的数据会报空指针
        comment.setParentComment(null);
    }
    comment.setCreateTime(new Date());//初始化数据
    commentRepository.save(comment);//保存操作
}

        private List<Comment> eachComment(List<Comment> comments){

      2. combineChildren遍历每个回复，将临时的list存入到list当中。

        private void combineChildren(List<Comment> comments){
    for (Comment comment: comments) {//遍历所有父节点
        List<Comment> replys1 = comment.getReplayComment();//获得一级子回复
        for(Comment reply:replys1){
            //调用方法，遍历子回复，找到子回复下的所有子回复，将其存入到临时list(tempReplys)当中
            recursively(reply);
        }
        comment.setReplayComment(tempReplys);//将临时的list存入到list
        tempReplys = new ArrayList<>();//清空临时list，这里的tempReplys是下面创建的全局list
    }
}

      3. recursively遍历所有二级回复，将自己和其子类加到临时集合中

        private List<Comment> tempReplys = new ArrayList<>();//用于存放父节点下面的所有子回复集合
private void recursively(Comment comment){
        tempReplys.add(comment);//先将一级回复加到里面，因为一级回复也是子回复
        if(comment.getReplayComment().size()>0){
            //如果还有子级回复
            List<Comment> replys = comment.getReplayComment();
            for(Comment reply : replys){
                tempReplys.add(reply);
                if (reply.getReplayComment().size()>0){
                    //子级回复还有子级回复
                    recursively(reply);//迭代
                }
            }
        }
    }
