        ModelAndView modelAndView = resolveErrorView(request, response, status, model);
        return (modelAndView == null ? new ModelAndView("error", model) : modelAndView);

    @ResponseBody    //дә§з”ҹjsonж•°жҚ®пјҢе…¶д»–е®ўжҲ·з«ҜжқҘеҲ°иҝҷдёӘж–№жі•еӨ„зҗҶпјӣ

                isIncludeStackTrace(request, MediaType.ALL));
        HttpStatus status = getStatus(request);
        return new ResponseEntity<Map<String, Object>>(body, status);
    }

вҖӢ дјҳзӮ№пјҡз®ҖеҚ•гҖҒдҫҝжҗәпјӣ

вҖӢ зјәзӮ№пјҡй»ҳи®ӨдёҚж”ҜжҢҒJSPгҖҒдјҳеҢ–е®ҡеҲ¶жҜ”иҫғеӨҚжқӮпјҲдҪҝз”Ёе®ҡеҲ¶еҷЁгҖҗServerPropertiesгҖҒиҮӘе®ҡд№үEmbeddedServletContainerCustomizerгҖ‘пјҢиҮӘе·ұзј–еҶҷеөҢе…ҘејҸServletе®№еҷЁзҡ„еҲӣе»әе·ҘеҺӮгҖҗEmbeddedServletContainerFactoryгҖ‘пјүпјӣ
